23pub fn validate_agents_have_linked_devices(
24    agents_with_chain_tops: &Vec<(AgentPubKey, ActionHash)>,
25    linked_devices_integrity_zome_name: ZomeName,
26) -> ExternResult<ValidateCallbackResult> {
27    let dna_info = dna_info()?;
28
29    let Some(linked_devices_integrity_zome_index) = dna_info
30        .zome_names
31        .into_iter()
32        .position(|z| z.eq(&linked_devices_integrity_zome_name))
33    else {
34        return Ok(ValidateCallbackResult::Invalid(String::from(
35            "Unreachable: there is no 'linked_devices' integrity zome in this DNA",
36        )));
37    };
38
39    validate_agents_have_linked_devices_with_zome_index(
40        agents_with_chain_tops,
41        ZomeIndex::new(linked_devices_integrity_zome_index as u8),
42    )
43}

45pub fn validate_agents_have_linked_devices_with_zome_index(
46    agents_with_chain_tops: &Vec<(AgentPubKey, ActionHash)>,
47    linked_devices_integrity_zome_index: ZomeIndex,
48) -> ExternResult<ValidateCallbackResult> {
49    let proofs: Vec<LinkedDevicesProof> =
50        get_all_linked_devices_proofs(agents_with_chain_tops, linked_devices_integrity_zome_index)?
51            .into_iter()
52            .collect();
53
54    for (agent_a, _chain_top) in agents_with_chain_tops {
55        for (agent_b, _) in agents_with_chain_tops {
56            if agent_a.eq(&agent_b) {
57                continue;
58            };
59
60            let devices_are_linked = are_agents_linked(&agent_a, &agent_b, &proofs);
61            if !devices_are_linked {
62                return Ok(ValidateCallbackResult::Invalid(format!(
63                    "Agents are haven't linked devices: {agent_a} {agent_b}"
64                )));
65            }
66        }
67    }
68
69    Ok(ValidateCallbackResult::Valid)
70}

72fn get_all_linked_devices_proofs(
73    agents_with_chain_tops: &Vec<(AgentPubKey, ActionHash)>,
74    linked_devices_integrity_zome_index: ZomeIndex,
75) -> ExternResult<BTreeSet<LinkedDevicesProof>> {
76    let mut result = BTreeSet::new();
77    for (agent, chain_top) in agents_with_chain_tops {
78        let activity = must_get_agent_activity(agent.clone(), ChainFilter::new(chain_top.clone()))?;
79        let tags = filter_agent_to_linked_devices_tags(
80            &activity,
81            linked_devices_integrity_zome_index.clone(),
82        )?;
83
84        for tag in tags {
85            for proof in tag.0 {
86                result.insert(proof);
87            }
88        }
89    }
90    Ok(result)
91}

119pub fn validate_agent_has_linked_device_with_zome_index(
120    agent: &AgentPubKey,
121    chain_top: &ActionHash,
122    linked_device: &AgentPubKey,
123    linked_devices_integrity_zome_index: ZomeIndex,
124) -> ExternResult<ValidateCallbackResult> {
125    let activity = must_get_agent_activity(agent.clone(), ChainFilter::new(chain_top.clone()))?;
126    let linked = has_linked_device(
127        agent,
128        &activity,
129        linked_device.clone(),
130        linked_devices_integrity_zome_index,
131    )?;
132    match linked {
133        true => Ok(ValidateCallbackResult::Valid),
134        false => Ok(ValidateCallbackResult::Invalid(format!(
135            "Agent has not linked the given device at the given chain top"
136        ))),
137    }
138}

140/** Utility functions */
141fn filter_agent_to_linked_devices_tags(
142    activity: &Vec<RegisterAgentActivity>,
143    linked_devices_integrity_zome_index: ZomeIndex,
144) -> ExternResult<Vec<AgentToLinkedDevicesLinkTag>> {
145    let agent_to_linked_devices_tags: Vec<AgentToLinkedDevicesLinkTag> = activity
146        .iter()
147        .filter_map(|activity| match &activity.action.hashed.content {
148            Action::CreateLink(create_link) => Some(create_link.clone()),
149            _ => None,
150        })
151        .filter(|create_link| {
152            create_link.zome_index == linked_devices_integrity_zome_index
153                && create_link.link_type.0 == AGENT_TO_LINKED_DEVICES_LINK_INDEX
154        })
155        .map(|create_link| {
156            let bytes =
157                SerializedBytes::from(UnsafeBytes::from(create_link.tag.into_inner().clone()));
158            let tag =
159                AgentToLinkedDevicesLinkTag::try_from(bytes).map_err(|err| wasm_error!(err))?;
160            Ok(tag)
161        })
162        .collect::<ExternResult<Vec<AgentToLinkedDevicesLinkTag>>>()?;
163    Ok(agent_to_linked_devices_tags)
164}

166fn has_linked_device(
167    agent: &AgentPubKey,
168    activity: &Vec<RegisterAgentActivity>,
169    linked_device: AgentPubKey,
170    linked_devices_integrity_zome_index: ZomeIndex,
171) -> ExternResult<bool> {
172    let agent_to_linked_devices_tags: Vec<AgentToLinkedDevicesLinkTag> =
173        filter_agent_to_linked_devices_tags(activity, linked_devices_integrity_zome_index)?;
174
175    let agent_to_linked_device = agent_to_linked_devices_tags
176        .into_iter()
177        .find(|tag| are_agents_linked(&agent, &linked_device, &tag.0));
178
179    Ok(agent_to_linked_device.is_some())
180}
