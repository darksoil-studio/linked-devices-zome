9pub fn query_my_linked_devices_agents() -> ExternResult<Vec<AgentPubKey>> {
10    let my_linked_devices_links = query_my_linked_devices(())?;
11
12    let my_linked_devices = my_linked_devices_links
13        .iter()
14        .map(|link| match link.target.clone().into_agent_pub_key() {
15            Some(pubkey) => Ok(pubkey),
16            None => Err(wasm_error!(WasmErrorInner::Guest(format!(
17                "Unreachable: an AgentToLinkedDevices link did not point to an AgentPubKey"
18            )))),
19        })
20        .collect::<ExternResult<Vec<AgentPubKey>>>()?;
21    Ok(my_linked_devices)
22}

24#[hdk_extern]
25pub fn query_my_linked_devices() -> ExternResult<Vec<Link>> {
26    let filter = ChainQueryFilter::new().action_type(ActionType::CreateLink);
27    let records = query(filter)?;
28
29    let links = records
30        .into_iter()
31        .map(|record| match record.action() {
32            Action::CreateLink(create_link) => Ok(create_link_to_link(
33                record.action_address().clone(),
34                create_link.clone(),
35            )),
36            _ => Err(wasm_error!(WasmErrorInner::Guest(format!(
37                "Unreachable: CreateLink query filter returned records that are not CreateLinks"
38            )))),
39        })
40        .collect::<ExternResult<Vec<Link>>>()?;
41
42    let linked_devices_links: Vec<Link> = links
43        .into_iter()
44        .filter(
45            |link| match LinkTypes::from_type(link.zome_index, link.link_type) {
46                Ok(Some(LinkTypes::AgentToLinkedDevices)) => true,
47                _ => false,
48            },
49        )
50        .collect();
51
52    Ok(linked_devices_links)
53}

55fn create_link_to_link(action_hash: ActionHash, create_link: CreateLink) -> Link {
56    Link {
57        author: create_link.author,
58        base: create_link.base_address,
59        target: create_link.target_address,
60        tag: create_link.tag,
61        timestamp: create_link.timestamp,
62        zome_index: create_link.zome_index,
63        link_type: create_link.link_type,
64
65        create_link_hash: action_hash,
66    }
