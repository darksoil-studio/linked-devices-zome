30pub fn action_hash(op: &Op) -> &ActionHash {
31    match op {
32        Op::StoreRecord(StoreRecord { record }) => record.action_address(),
33        Op::StoreEntry(StoreEntry { action, .. }) => &action.hashed.hash,
34        Op::RegisterUpdate(RegisterUpdate { update, .. }) => &update.hashed.hash,
35        Op::RegisterDelete(RegisterDelete { delete, .. }) => &delete.hashed.hash,
36        Op::RegisterAgentActivity(RegisterAgentActivity { action, .. }) => &action.hashed.hash,
37        Op::RegisterCreateLink(RegisterCreateLink { create_link }) => &create_link.hashed.hash,
38        Op::RegisterDeleteLink(RegisterDeleteLink { delete_link, .. }) => &delete_link.hashed.hash,
39    }
40}

42// This is the unified validation callback for all entries and link types in this integrity zome
43// Below is a match template for all of the variants of `DHT Ops` and entry and link types
44//
45// Holochain has already performed the following validation for you:
46// - The action signature matches on the hash of its content and is signed by its author
47// - The previous action exists, has a lower timestamp than the new action, and incremented sequence number
48// - The previous action author is the same as the new action author
49// - The timestamp of each action is after the DNA's origin time
50// - AgentActivity authorities check that the agent hasn't forked their chain
51// - The entry hash in the action matches the entry content
52// - The entry type in the action matches the entry content
53// - The entry size doesn't exceed the maximum entry size (currently 4MB)
54// - Private entry types are not included in the Op content, and public entry types are
55// - If the `Op` is an update or a delete, the original action exists and is a `Create` or `Update` action
56// - If the `Op` is an update, the original entry exists and is of the same type as the new one
57// - If the `Op` is a delete link, the original action exists and is a `CreateLink` action
58// - Link tags don't exceed the maximum tag size (currently 1KB)
59// - Countersigned entries include an action from each required signer
60//
61// You can read more about validation here: https://docs.rs/hdi/latest/hdi/index.html#data-validation
62
63#[hdk_extern]
64pub fn validate(op: Op) -> ExternResult<ValidateCallbackResult> {
65    match op.flattened::<(), LinkTypes>()? {
66        FlatOp::StoreEntry(store_entry) => match store_entry {
67            OpEntry::CreateEntry { .. } => Ok(ValidateCallbackResult::Invalid(
68                "There are no entry types in this integrity zome".to_string(),
69            )),
70            OpEntry::UpdateEntry { .. } => Ok(ValidateCallbackResult::Invalid(
71                "There are no entry types in this integrity zome".to_string(),
72            )),
73            _ => Ok(ValidateCallbackResult::Valid),
74        },
75        FlatOp::RegisterUpdate(update_entry) => match update_entry {
76            OpUpdate::Entry { .. } => Ok(ValidateCallbackResult::Invalid(
77                "There are no entry types in this integrity zome".to_string(),
78            )),
79            _ => Ok(ValidateCallbackResult::Valid),
80        },
81        FlatOp::RegisterDelete(delete_entry) => {
82            let record = must_get_valid_record(delete_entry.action.deletes_address)?;
83            let Action::Create(create) = record.action() else {
84                return Ok(ValidateCallbackResult::Invalid(
85                    "There are no entry types in this integrity zome".to_string(),
86                ));
87            };
88            if let EntryType::CapGrant = create.entry_type {
89                return Ok(ValidateCallbackResult::Valid);
90            }
91            Ok(ValidateCallbackResult::Invalid(
92                "There are no entry types in this integrity zome".to_string(),
93            ))
94        }
95        FlatOp::RegisterCreateLink {
96            link_type,
97            base_address,
98            target_address,
99            tag,
100            action,
101        } => match link_type {
102            LinkTypes::AgentToLinkedDevices => validate_create_link_agent_to_linked_devices(
103                action_hash(&op).clone(),
104                action,
105                base_address,
106                target_address,
107                tag,
108            ),
109        },
110        FlatOp::RegisterDeleteLink {
111            link_type,
112            base_address,
113            target_address,
114            tag,
115            original_action,
116            action,
117        } => match link_type {
118            LinkTypes::AgentToLinkedDevices => validate_delete_link_agent_to_linked_devices(
119                action,
120                original_action,
121                base_address,
122                target_address,
123                tag,
124            ),
125        },
126        FlatOp::StoreRecord(store_record) => {
127            match store_record {
128                // Complementary validation to the `StoreEntry` Op, in which the record itself is validated
129                // If you want to optimize performance, you can remove the validation for an entry type here and keep it in `StoreEntry`
130                // Notice that doing so will cause `must_get_valid_record` for this record to return a valid record even if the `StoreEntry` validation failed
131                OpRecord::CreateEntry { action, .. } => {
132                    if let EntryType::CapGrant = action.entry_type {
133                        return Ok(ValidateCallbackResult::Valid);
134                    }
135                    Ok(ValidateCallbackResult::Invalid(
136                        "There are no entry types in this integrity zome".to_string(),
137                    ))
138                }
139                // Complementary validation to the `RegisterUpdate` Op, in which the record itself is validated
140                // If you want to optimize performance, you can remove the validation for an entry type here and keep it in `StoreEntry` and in `RegisterUpdate`
141                // Notice that doing so will cause `must_get_valid_record` for this record to return a valid record even if the other validations failed
142                OpRecord::UpdateEntry { .. } => Ok(ValidateCallbackResult::Invalid(
143                    "There are no entry types in this integrity zome".to_string(),
144                )),
145                // Complementary validation to the `RegisterDelete` Op, in which the record itself is validated
146                // If you want to optimize performance, you can remove the validation for an entry type here and keep it in `RegisterDelete`
147                // Notice that doing so will cause `must_get_valid_record` for this record to return a valid record even if the `RegisterDelete` validation failed
148                OpRecord::DeleteEntry {
149                    original_action_hash,
150                    ..
151                } => {
152                    let record = must_get_valid_record(original_action_hash)?;
153                    let Action::Create(create) = record.action() else {
154                        return Ok(ValidateCallbackResult::Invalid(
155                            "There are no entry types in this integrity zome".to_string(),
156                        ));
157                    };
158                    if let EntryType::CapGrant = create.entry_type {
159                        return Ok(ValidateCallbackResult::Valid);
160                    }
161                    Ok(ValidateCallbackResult::Invalid(
162                        "There are no entry types in this integrity zome".to_string(),
163                    ))
164                }
165                // Complementary validation to the `RegisterCreateLink` Op, in which the record itself is validated
166                // If you want to optimize performance, you can remove the validation for an entry type here and keep it in `RegisterCreateLink`
167                // Notice that doing so will cause `must_get_valid_record` for this record to return a valid record even if the `RegisterCreateLink` validation failed
168                OpRecord::CreateLink {
169                    base_address,
170                    target_address,
171                    tag,
172                    link_type,
173                    action,
174                } => match link_type {
175                    LinkTypes::AgentToLinkedDevices => {
176                        validate_create_link_agent_to_linked_devices(
177                            action_hash(&op).clone(),
178                            action,
179                            base_address,
180                            target_address,
181                            tag,
182                        )
183                    }
184                },
185                // Complementary validation to the `RegisterDeleteLink` Op, in which the record itself is validated
186                // If you want to optimize performance, you can remove the validation for an entry type here and keep it in `RegisterDeleteLink`
187                // Notice that doing so will cause `must_get_valid_record` for this record to return a valid record even if the `RegisterDeleteLink` validation failed
188                OpRecord::DeleteLink {
189                    original_action_hash,
190                    base_address,
191                    action,
192                } => {
193                    let record = must_get_valid_record(original_action_hash)?;
194                    let create_link = match record.action() {
195                        Action::CreateLink(create_link) => create_link.clone(),
196                        _ => {
197                            return Ok(ValidateCallbackResult::Invalid(
198                                "The action that a DeleteLink deletes must be a CreateLink"
199                                    .to_string(),
200                            ));
201                        }
202                    };
203                    let link_type = match LinkTypes::from_type(
204                        create_link.zome_index,
205                        create_link.link_type,
206                    )? {
207                        Some(lt) => lt,
208                        None => {
209                            return Ok(ValidateCallbackResult::Valid);
210                        }
211                    };
212                    match link_type {
213                        LinkTypes::AgentToLinkedDevices => {
214                            validate_delete_link_agent_to_linked_devices(
215                                action,
216                                create_link.clone(),
217                                base_address,
218                                create_link.target_address,
219                                create_link.tag,
220                            )
221                        }
222                    }
223                }
224                OpRecord::CreatePrivateEntry { .. } => Ok(ValidateCallbackResult::Valid),
225                OpRecord::UpdatePrivateEntry { .. } => Ok(ValidateCallbackResult::Valid),
226                OpRecord::CreateCapClaim { .. } => Ok(ValidateCallbackResult::Valid),
227                OpRecord::CreateCapGrant { .. } => Ok(ValidateCallbackResult::Valid),
228                OpRecord::UpdateCapClaim { .. } => Ok(ValidateCallbackResult::Valid),
229                OpRecord::UpdateCapGrant { .. } => Ok(ValidateCallbackResult::Valid),
230                OpRecord::Dna { .. } => Ok(ValidateCallbackResult::Valid),
231                OpRecord::OpenChain { .. } => Ok(ValidateCallbackResult::Valid),
232                OpRecord::CloseChain { .. } => Ok(ValidateCallbackResult::Valid),
233                OpRecord::InitZomesComplete { .. } => Ok(ValidateCallbackResult::Valid),
234                _ => Ok(ValidateCallbackResult::Valid),
235            }
236        }
237        FlatOp::RegisterAgentActivity(agent_activity) => match agent_activity {
238            OpActivity::CreateAgent { agent, action } => {
239                let previous_action = must_get_action(action.prev_action)?;
240                match previous_action.action() {
241                        Action::AgentValidationPkg(
242                            AgentValidationPkg { membrane_proof, .. },
243                        ) => validate_agent_joining(agent, membrane_proof),
244                        _ => {
245                            Ok(
246                                ValidateCallbackResult::Invalid(
247                                    "The previous action for a `CreateAgent` action must be an `AgentValidationPkg`"
248                                        .to_string(),
249                                ),
250                            )
251                        }
252                    }
253            }
254            _ => Ok(ValidateCallbackResult::Valid),
255        },
256    }
